'⏳' : '✅'; // עיצוב גוף ההודעה - נקי, ברור ומקצועי const messageBody = ` *📦 הזמנה חדשה במערכת* --------------------------- 👤 *לקוח:* ${order.customerName} 🔗 *מקור:* ${order.source} 🆔 *מספר הזמנה:* #ַ${order.orderId} ${statusEmoji} *סטטוס:* ${order.status} 📅 *דוח בוקר:* זמין כעת בערוץ. --------------------------- _הודעה זו נוצרה אוטומטית ע"י SabanOS_ `; try { const response = await axios.post( WHATSAPP_API_URL, { messaging_product: "whatsapp", recipient_type: "individual", to: "CHANNEL_ID_OR_PHONE_NUMBER", // מזהה הערוץ type: "text", text: { preview_url: true, body: messageBody } }, { headers: { 'Authorization': `Bearer ${ACCESS_TOKEN}`, 'Content-Type': 'application/json' } } ); console.log('✅ הכרטיס נשלח בהצלחה:', response.data); } catch (error) { console.error('❌ שגיאה בשליחת הכרטיס:', error.response ? error.response.data : error.message); } } // דוגמה להפעלה ברגע שמתקבלת הזמנה מזבולון-עדירן const newOrder = { customerName: 'זבולון-עדירן', source: 'קבוצת הזמנות', orderId: '88293', status: 'פתוח' }; sendOrderCard(newOrder);
